# Add slug column for URL-friendly names

def create_slug(text):
    if pd.isna(text):
        return None
    return re.sub(r'[^a-z0-9]+', '-', text.lower()).strip('-')

df["slug_L1"] = df["Category_L1"].apply(create_slug)
df["slug_L2"] = df["Category_L2"].apply(create_slug)

for idx, row in df.iterrows():
    if pd.notna(row["Category_L2"]):
        # Find parent L1
        parent = df[(df["Category_L1"] == row["Category_L1"]) & (df["Category_L2"].isna()) & (df["Category_L3"].isna())]
        if len(parent) > 0:
            df.at[idx, "parent_id"] = parent.iloc[0]["category_id"]
    
    if pd.notna(row["Category_L3"]) and pd.notna(row["Category_L2"]):
        # Find parent L2
        parent = df[(df["Category_L1"] == row["Category_L1"]) & (df["Category_L2"] == row["Category_L2"]) & (df["Category_L3"].isna())]
        if len(parent) > 0:
            df.at[idx, "parent_id"] = parent.iloc[0]["category_id"]
    
    if pd.notna(row["Category_L4"]) and pd.notna(row["Category_L3"]):
        # Find parent L3
        parent = df[(df["Category_L1"] == row["Category_L1"]) & (df["Category_L2"] == row["Category_L2"]) & (df["Category_L3"] == row["Category_L3"]) & (df["Category_L4"].isna())]
        if len(parent) > 0:
            df.at[idx, "parent_id"] = parent.iloc[0]["category_id"]

    if pd.notna(row["Category_L5"]):
        return 5
    elif pd.notna(row["Category_L4"]):
        return 4
    elif pd.notna(row["Category_L3"]):
        return 3
    elif pd.notna(row["Category_L2"]):
        return 2
    else:
        return 1

        parts.append(row["Category_L4"])
    if pd.notna(row["Category_L5"]):

    return " > ".join(parts)
